3/// 将OHLCV数据转换为字节数组
4pub fn ohlcv_to_bytes(data: &[OHLCVData]) -> Vec<Vec<u8>> {
5    let mut result = Vec::new();
6    
7    // Epoch列
8    let epochs: Vec<i64> = data.iter().map(|d| d.epoch).collect();
9    result.push(epochs.into_iter().map(|e| e.to_le_bytes()).flatten().collect());
10    
11    // Open列
12    let opens: Vec<f32> = data.iter().map(|d| d.open).collect();
13    result.push(opens.into_iter().map(|o| o.to_le_bytes()).flatten().collect());
14    
15    // High列
16    let highs: Vec<f32> = data.iter().map(|d| d.high).collect();
17    result.push(highs.into_iter().map(|h| h.to_le_bytes()).flatten().collect());
18    
19    // Low列
20    let lows: Vec<f32> = data.iter().map(|d| d.low).collect();
21    result.push(lows.into_iter().map(|l| l.to_le_bytes()).flatten().collect());
22    
23    // Close列
24    let closes: Vec<f32> = data.iter().map(|d| d.close).collect();
25    result.push(closes.into_iter().map(|c| c.to_le_bytes()).flatten().collect());
26    
27    // Volume列
28    let volumes: Vec<f32> = data.iter().map(|d| d.volume).collect();
29    result.push(volumes.into_iter().map(|v| v.to_le_bytes()).flatten().collect());
30    
31    result
32}

34/// 从字节数组创建NumpyDataset
35pub fn create_numpy_dataset_from_ohlcv(data: &[OHLCVData]) -> NumpyDataset {
36    let column_data = ohlcv_to_bytes(data);
37    
38    NumpyDataset {
39        column_types: vec!["i8".to_string(), "f4".to_string(), "f4".to_string(), 
40                          "f4".to_string(), "f4".to_string(), "f4".to_string()],
41        column_names: vec!["Epoch".to_string(), "Open".to_string(), "High".to_string(),
42                          "Low".to_string(), "Close".to_string(), "Volume".to_string()],
43        column_data,
44        length: data.len() as i32,
45    }
46}
